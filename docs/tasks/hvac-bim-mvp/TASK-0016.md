```typescript
import { useScene } from '@pascal-app/core'
import { useViewer } from '@pascal-app/viewer'
import type { HvacZoneNode } from '@pascal-app/core/schema/nodes/hvac-zone'

export function HvacZonePanel() {
  const selectedIds = useViewer((s) => s.selectedIds)
  const zoneId = selectedIds[0] // 単一選択前提
  const node = useScene((s) => s.nodes[zoneId]) as HvacZoneNode | undefined
  const updateNode = useScene((s) => s.updateNode)

  if (!node || node.type !== 'hvac_zone') return null

  return (
    <div className="hvac-zone-panel">
      <h3>ゾーンプロパティ</h3>
      <BasicInfoSection node={node} onUpdate={updateNode} />
      <DesignConditionsSection node={node} onUpdate={updateNode} />
      <PerimeterSection node={node} />
    </div>
  )
}
```

### 2. 🔵 ゾーン基本情報表示/編集（zoneName, usage, floorArea, ceilingHeight, occupantDensity）

```typescript
function BasicInfoSection({
  node,
  onUpdate,
}: {
  node: HvacZoneNode
  onUpdate: (id: string, data: Partial<HvacZoneNode>) => void
}) {
  return (
    <section>
      <label>ゾーン名</label>
      <input
        value={node.zoneName}
        onChange={(e) => onUpdate(node.id, { zoneName: e.target.value })}
      />

      <label>用途</label>
      <select
        value={node.usage}
        onChange={(e) => onUpdate(node.id, { usage: e.target.value })}
      >
        <option value="office_general">一般オフィス</option>
        <option value="conference">会議室</option>
        <option value="reception">受付/ロビー</option>
        <option value="office_server">サーバー室</option>
        <option value="corridor">廊下</option>
      </select>

      <label>床面積 (m²)</label>
      <span>{node.floorArea.toFixed(1)}</span> {/* 読み取り専用 */}

      <label>天井高 (m)</label>
      <input
        type="number"
        value={node.ceilingHeight}
        onChange={(e) => onUpdate(node.id, { ceilingHeight: Number(e.target.value) })}
      />

      <label>在室人員密度 (人/m²)</label>
      <input
        type="number"
        value={node.occupantDensity}
        step={0.01}
        onChange={(e) => onUpdate(node.id, { occupantDensity: Number(e.target.value) })}
      />
    </section>
  )
}
```

- floorAreaはゾーン境界から自動算出のため読み取り専用
- usageの変更はdirtyマーク → LoadCalcSystem再計算をトリガー
- ceilingHeight, occupantDensityは数値入力

```typescript
function PerimeterSection({ node }: { node: HvacZoneNode }) {
  const segments = node.perimeterSegments ?? []

  if (segments.length === 0) {
    return (
      <section>
        <h4>外皮条件</h4>
        <p>未設定（PerimeterEditToolで入力してください）</p>
      </section>
    )
  }

  return (
    <section>
      <h4>外皮条件</h4>
      <table>
        <thead>
          <tr>
            <th>方位</th>
            <th>壁面積 (m²)</th>
            <th>ガラス面積比</th>
          </tr>
        </thead>
        <tbody>
          {segments.map((seg, i) => (
            <tr key={i}>
              <td>{seg.orientation}</td>
              <td>{seg.wallArea.toFixed(1)}</td>
              <td>{(seg.glazingRatio * 100).toFixed(0)}%</td>
            </tr>
          ))}
        </tbody>
      </table>
    </section>
  )
}
```

- テーブル形式で一覧表示
- glazingRatioはパーセント表記（×100）
- 未設定時は誘導メッセージ表示

計算結果の詳細表示。

```typescript
function CalcResultDetails({ result }: { result: CalcResult }) {
  return (
    <>
      <section>
        <h4>負荷サマリー</h4>
        <dl>
          <dt>冷房負荷</dt>
          <dd>{formatLoad(result.coolingLoad)}</dd>
          <dt>暖房負荷</dt>
          <dd>{formatLoad(result.heatingLoad)}</dd>
          <dt>必要風量</dt>
          <dd>{result.requiredAirflow.toLocaleString()} m³/h</dd>
        </dl>
      </section>

      <section>
        <h4>負荷内訳</h4>
        <dl>
          <dt>内部負荷</dt>
          <dd>{formatLoad(result.internalLoad)}</dd>
          <dt>外皮負荷</dt>
          <dd>{formatLoad(result.envelopeLoad)}</dd>
        </dl>
      </section>

      {result.perimeterLoadBreakdown.length > 0 && (
        <section>
          <h4>方位別外皮負荷</h4>
          <table>
            <thead>
              <tr>
                <th>方位</th>
                <th>負荷 (W)</th>
                <th>割合</th>
              </tr>
            </thead>
            <tbody>
              {result.perimeterLoadBreakdown.map((item) => (
                <tr key={item.orientation}>
                  <td>{item.orientation}</td>
                  <td>{formatLoad(item.load)}</td>
                  <td>{item.percentage.toFixed(1)}%</td>
                </tr>
              ))}
            </tbody>
          </table>
        </section>
      )}
    </>
  )
}

function formatLoad(watts: number): string {
  if (watts >= 1000) {
    return `${(watts / 1000).toFixed(1)} kW`
  }
  return `${Math.round(watts)} W`
}
```
